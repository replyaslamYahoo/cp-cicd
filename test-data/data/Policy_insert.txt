Map<String, Id> personAccountByLastName = new Map<String, Id>();
for (Account a : [
    SELECT Id, LastName
    FROM Account
    WHERE IsPersonAccount = true
      AND LastName IN ('Test Customer', 'Sample Customer', 'Demo Customer')
]) {
    personAccountByLastName.put(a.LastName, a.Id);
}

Map<Id, Id> contactIdByPersonAccountId = new Map<Id, Id>();
for (Contact c : [SELECT Id, AccountId FROM Contact WHERE AccountId IN :personAccountByLastName.values()]) {
    contactIdByPersonAccountId.put(c.AccountId, c.Id);
}

Map<String, Id> carrierByName = new Map<String, Id>();
for (Account a : [SELECT Id, Name FROM Account WHERE RecordType.DeveloperName = 'Carrier' AND Name IN ('QBE Insurance', 'Allianz', 'Suncorp', 'IAG')]) {
    carrierByName.put(a.Name, a.Id);
}

Id getContactId(String lastName) {
    Id personAccId = personAccountByLastName.get(lastName);
    if (personAccId == null) return null;
    return contactIdByPersonAccountId.get(personAccId);
}
